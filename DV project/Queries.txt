TAT in Days = 
    IF(
        ISBLANK('MasterData'[ValidatedDate]), 
        0, 
        DATEDIFF('MasterData'[AddedDate], 'MasterData'[ValidatedDate], DAY)
    )


Total TAT = 
    SUM('MasterData'[Tat in days])


Average TAT = 
    AVERAGE('MasterData'[TAT in Days])

    )
